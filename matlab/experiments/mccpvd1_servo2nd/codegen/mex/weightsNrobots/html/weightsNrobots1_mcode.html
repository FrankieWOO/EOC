1function [ res ] = weightsNrobots()
  2%format long
  3%---- user specification ----%
  4target = 0.5;
  5T = 2;
  6dt = 0.02;
  7N = T/dt + 1;
  8position0 = 0;
  9x0 = zeros(6,1); 
 10x0(1) = position0;
 11x0(3) = position0;
 12x0(5) = 0;
 13
 14weights = linspace(0.0001,0.005,51);
 15Nw = 51;
 16ratio_loads = [0, 0.25, 0.5, 0.75, 1];
 17Nr = length(ratio_loads);
 18robot_params = cell(Nr);
 19robot_models = cell(Nr);
 20for i=1:Nr
 21    robot_params{i}.ratio_load = ratio_loads(i); 
 22    robot_models{i} = Mccpvd1dofModel();
 23    robot_models{i}.actuator = ActMccpvd(robot_params{i});
 24
 25end
 26
 27task_params = cell(Nw,Nr);
 28task_params2 = cell(Nw,Nr);
 29tasks = cell(Nw,Nr);
 30tasks2 = cell(Nw,Nr);
 31for i = 1:Nw
 32    for j = 1:Nr 
 33        task_params{i,j}.w = weights(i);
 34        task_params{i,j}.epsilon = 0;
 35        task_params{i,j}.T = T;
 36        task_params{i,j}.dt = dt;
 37        task_params{i,j}.target = target;
 38        task_params{i,j}.fd = 1; % use finite difference or not
 39        task_params{i,j}.x0 = x0;
 40        
 41        task_params{i,j}.f = @(x,u)robot_models{j}.dynamics_with_jacobian(x,u);
 42        
 43        task_params2{i,j}=task_params{i,j};
 44        task_params2{i,j}.w = task_params{i,j}.w*100;
 45        
 46        tasks{i,j} = mccpvd1_reach(robot_models{j}, task_params{i,j});
 47        tasks2{i,j} = mccpvd1_reach(robot_models{j}, task_params2{i,j});
 48        
 49        task_params{i,j}.j_netelec = @(x,u,t)tasks{i,j}.j_netelec(x,u,t);
 50        task_params{i,j}.j_elec = @(x,u,t)tasks{i,j}.j_elec(x,u,t);
 51        
 52        task_params2{i,j}.j_noutmech = @(x,u,t)tasks2{i,j}.j_noutmech(x,u,t);
 53        task_params2{i,j}.j_outmech = @(x,u,t)tasks2{i,j}.j_outmech(x,u,t);
 54        
 55    end
 56end
 57
 58
 59
 60%%
 61opt_param = [];
 62opt_param.umax = robot_models{1}.umax;
 63opt_param.umin = robot_models{1}.umin;
 64opt_param.lambda_init = 0.01;
 65opt_param.lambda_max  = 0.5;
 66opt_param.iter_max = 100;
 67opt_param.online_plotting = 0;
 68opt_param.online_printing = 1;
 69opt_param.dcost_converge = 10^-6;
 70opt_param.solver = 'rk4';
 71opt_param.target = target;
 72opt_param.T = T;
 73opt_param.dt = dt;
 74% u0 can be full command sequence or just initial point
 75u0 = [target; 0; 0];
 76%u0 = [0; 0.1; 0];
 77results = cell(Nw,Nr);
 78for i = 1:Nw
 79   for j =1:Nr
 80        results{i,j}.result_elec_rege0 = [];
 81        results{i,j}.result_elec_rege1 = [];
 82        results{i,j}.result_outmech_rege0=[];
 83        results{i,j}.result_outmech_rege1=[];
 84   end
 85end
 86
 87parfor i = 1:Nw
 88    for j = 1:Nr
 89%result = ILQRController.ilqr(f, j, dt, N, x0, u0, opt_param);
 90        results{i,j}.result_elec_rege0 = ILQRController.ilqr(task_params{i,j}.f, task_params{i,j}.j_elec, dt, N, x0, u0, opt_param);
 91        results{i,j}.result_elec_rege1 = ILQRController.ilqr(task_params{i,j}.f, task_params{i,j}.j_netelec, dt, N, x0, u0, opt_param);
 92        results{i,j}.result_outmech_rege0 = ILQRController.ilqr(task_params2{i,j}.f, task_params2{i,j}.j_outmech, dt, N, x0, u0, opt_param);
 93        results{i,j}.result_outmech_rege1 = ILQRController.ilqr(task_params2{i,j}.f, task_params2{i,j}.j_noutmech, dt, N, x0, u0, opt_param);
 94        
 95    end
 96end

 98
 99%%
100res.robot_models = robot_models;
